#List available disks and format it
fdisk -l					#list disks
# format partitions
fdisk /dev/sd*				#format drive
o						#blank out drive
n						#use default first sector				
Enter						#default primaty partition
Enter, Enter					#default partition number and first sector
+500M					#boot partition size
n						#use default first sector				
Enter						#default primaty partition
Enter, Enter					#default partition number and first sector
+**GB					            #swap size Ex: For 8GB 12GB of swap
type, Enter, 82				#more swappines
n						#use default first sector				
Enter						#default primaty partition
Enter, Enter					#default partition number and first sector
+**GB						#root part. size (Recommendation: 30GB at least)
n						#use default first sector				
Enter						#default primaty partition
Enter, Enter					#default partition number and first sector
Enter						#home partition size
w						#write changes
fdisk -l					#check if succesfull

# format partitions
mkfs.ext4 /dev/sd*1			#format boot partition as ext4
mkfs.ext4 /dev/sd*3			#format root partition as ext4
mkfs.ext4 /dev/sd*4			#format home partition as ext4
mkswap /dev/sd*2				#format swap
swapon /dev/sd*2				#activate swap
# mount root and home partitions
mount /dev/sd*3 /mnt
mkdir /mnt/boot
mkdir /mnt/home
mount /dev/sd*1 /mnt/boot
mount /dev/sd*4 /mnt/home
# check internet connection
ping -c 5 8.8.8.8
# install base system, generate fstab, and enter your install
pacstrap -i /mnt base base-devel		# install the base and base-devel system
genfstab -U -p /mnt >> /mnt/etc/fstab  	#generates fstab
arch-chroot /mnt			# enters your installation 
nano /etc/fstab			#add discard to /sd*1 /sd*3 /sd*4 if you have SSD drive
# install GRUB and kernel headers (linux-lts optional)
pacman -S grub-bios linux-headers linux-lts linux-lts-headers
# install linux firmware (includes wifi interfaces)
pacman -S linux-firmware
# install wireless tools (if needed)
pacman -S wireless_tools wpa_supplicant dialog wpa_actiond

# edit pacman.conf to enable 32bit libraries
pacman -S nano					# if it’s not in the system already
nano /etc/pacman.conf				# uncomment [multilib] and line below
pacman -Syyu					# updates new libraries + upgrade system
# locale
nano /etc/locale.gen 				# uncomment your locale		
locale-gen						# generate locale
rm -f /etc/localtime					# removes the one generated by the system
ln -s /usr/share/zoneinfo/<Continent>/<City>  /etc/localtime  
hwclock --systohc –utc				#Syncs hardware clock to sys time
# set passwd for root
passwd 						# Set root password
# install grub and generate .cfg
grub-install --target=i386-pc --recheck /dev/sd*	# install grub menu on the selected drive
cp /usr/share/locale/en\@quot/LC_MESSAGES/grub.mo /boot/grub/locale/en.mo 
grub-mkconfig -o /boot/grub/grub.cfg		# Generates grub config
# install sudo and set sudo users
pacman -S sudo				# Install sudo in case not installed
usedadd -m -G wheel -s /bin/bash <username>	#Add user to /bin/bash
visudo  OR EDITOR=nano visudo (for non vim boomers) 
# Edit visudo file and uncomment %wheel group
passwd <username>					# Generate password for new user
# install networkmanager
pacman -S networkmanager network-manager-applet		#Install networkmanager
systemctl disable dhcpcd@<interfacename>.service		# Disable dhcpcd service
systemctl disable netctl-auto@<wifi-interface>.service		# Disable netctl
systemctl enable NetworkManager.service		# Enable Networkmanager at boot
# exit live cd
exit 							# exit live install
umount /mnt/home	
umount /mnt/boot 				# umount all drives
umount /mnt
reboot						#reboots (remove install media)				
# Log in
# hostname
hostnamectl set-hostname <hostname>		#Set hostname of your machine	
localectl set-locale LANG=”<en_US.UTF-8>”		# Set locale language
# install various things
pacman -S xf86-input-libinput 			# Install trackpoint or other input method drivers
pacman -S xorg xorg-server xorg-xinit  mesa		#xorg and mesa install
# intel # IF YOU HAVE INTEL GRAPHICS or AMD
pacman -S lib32-intel-dri lib32-mesa lib32-libgl		
# nvidia #IF YOU HAVE AN NVIDIA CARD
pacman S nvidia lib32nvidia-libgl nvidia-settings nvidia-utils lib32-nvidia-utils libvdpau lib32-libvdpau			
# Desktop envioronment (To use others lookup on Arch Wiki)
echo “exec <DE>” > ~/.xinitrc
# XFCE					
DE= startxfce4				
pacman -S xfce4 xfce4-goodies		
# install login service and fix for long boot time
pacman -S sddm sddm-kcm
systemctl enable sddm.service
pacman -S haveged
systemctl enable haveged.service					AND REBOOT!
